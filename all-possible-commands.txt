# === BEST PRACTICES ===

# 2. Wait 2-3 minutes between commands

# 4. Monitor invitation success rate

# === RATE LIMIT MANAGEMENT ===
# Morning Batch (100 invites)

# Wait 3 minutes
npm run invite "language:Python followers:>50" 100
# Wait 3 minutes
npm run invite "language:Java followers:>50" 100

# === SEARCH OPERATORS ===
# AND operator (space between terms)
npm run invite "language:JavaScript location:USA" 100

# OR operator
npm run invite "language:JavaScript OR language:TypeScript" 100

# NOT operator
npm run invite "language:JavaScript NOT language:TypeScript" 100

# === TIPS FOR SUCCESS ===
# 1. Mix different search criteria for diverse candidates
# 2. Use location-specific searches during target region's working hours
# 3. Adjust follower counts based on language/location popularity
# 4. Use quotes for exact matches
# 5. Combine multiple criteria for more targeted results

# === TROUBLESHOOTING ===
# If rate limited:
# 1. Check: npm run stats
# 2. Wait for reset time
# 3. Reduce batch size
# 4. Increase wait time between batches
